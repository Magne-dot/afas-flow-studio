Proxy Setup
One-time Cloudflare Worker deployment to solve CORS
Why a proxy is needed
Browsers block direct calls to the AFAS API from a webpage (CORS policy). A tiny Cloudflare Worker acts as a relay — your browser calls the Worker, the Worker calls AFAS, and sends the response back. It's free for up to 100,000 requests/day and takes about 5 minutes to set up.
Step by step deployment
1
Create a free Cloudflare account
Go to dash.cloudflare.com/sign-up — no credit card needed for Workers free tier.
2
Create a new Worker
Go to Workers & Pages → Create → Create Worker. Give it a name like afas-proxy. Click Deploy, then Edit code.
3
Paste this code into the Worker editor
export default {
  async fetch(request, env) {
    // Allow CORS preflight
    if (request.method === 'OPTIONS') {
      return corsResponse(new Response(null, { status: 204 }));
    }

    // Read JSON envelope from POST body
    let envelope;
    try {
      envelope = await request.json();
    } catch(e) {
      return corsResponse(new Response(
        JSON.stringify({ error: 'Invalid JSON body' }),
        { status: 400, headers: { 'Content-Type': 'application/json' } }
      ));
    }

    const { target, method, authorization, body } = envelope;

    if (!target) {
      return corsResponse(new Response(
        JSON.stringify({ error: 'Missing target in envelope' }),
        { status: 400, headers: { 'Content-Type': 'application/json' } }
      ));
    }

    // Forward the request to AFAS
    const headers = new Headers();
    headers.set('Authorization', authorization || '');
    headers.set('Content-Type', 'application/json');
    headers.set('Accept', 'application/json');

    const afasRes = await fetch(target, {
      method: method || 'GET',
      headers,
      body: body || undefined
    });

    const text = await afasRes.text();
    return corsResponse(new Response(text, {
      status: afasRes.status,
      headers: { 'Content-Type': 'application/json' }
    }));
  }
};

function corsResponse(res) {
  const r = new Response(res.body, res);
  r.headers.set('Access-Control-Allow-Origin', '*');
  r.headers.set('Access-Control-Allow-Headers', 'Content-Type');
  r.headers.set('Access-Control-Allow-Methods', 'POST, OPTIONS');
  return r;
}
